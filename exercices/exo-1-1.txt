En supposant les <?> du code ci-dessous remplacés par le code de format adéquat, indiquer ce
que va afficher le programme suivant ?

#include <stdio.h>
#include <stdlib.h>

#define A 1
#define B 2
#define AB 3
#define F(X, Y) X##Y

int main(void) {
    int n = AB;
    printf("%<?>\n", n);            // 3
    printf("%<?>\n", "AB");         // AB
    printf("%<?>\n", F(A, B));      // 3
    return EXIT_SUCCESS;
}
